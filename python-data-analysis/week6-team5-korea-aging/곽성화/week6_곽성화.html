# mysql 데이터 조회 함수
def sql_select(sql):
    conn = pymysql.connect(host='localhost', user='root', password='0000', db = 'team_project', charset='utf8')  # DB 접속
    curs = conn.cursor()       # 커서 생성
    curs.execute(sql)          # sql문 실행
    rows = curs.fetchall()     # 모든 데이터를 가져옴
    curs.close()               # 커서 종료
    conn.close()               # MySQL 연결 종료
    return rows

근무시간 및 보수 상승 등 퀄리티(질) 상승도 필요하다고 사료됨
